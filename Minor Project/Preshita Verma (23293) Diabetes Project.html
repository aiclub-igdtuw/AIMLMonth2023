import pandas as pd
import scipy.stats as stats

# Load the dataset into a pandas DataFrame
df = pd.read_csv('Diabetes.csv')  

# Split the data into two groups: individuals with and without diabetes
diabetes_positive = df[df['Outcome'] == 1]
diabetes_negative = df[df['Outcome'] == 0]

# Perform a hypothesis test (two-sample t-test) to compare the mean glucose levels
_, p_value = stats.ttest_ind(diabetes_positive['Glucose'], diabetes_negative['Glucose'], equal_var=False)

# Determine the significance level
alpha = 0.05

# Print the results
print("Hypothesis Test Results:")
print(f"P-value: {p_value}")
if p_value < alpha:
    print("There is a significant difference in mean glucose levels for individuals with and without diabetes.")
else:
    print("There is no significant difference in mean glucose levels for individuals with and without diabetes.")

import pandas as pd
import scipy.stats as stats

# Load the dataset into a pandas DataFrame
df = pd.read_csv('Diabetes.csv')

# Split the data into two groups: individuals with and without diabetes
diabetes_positive = df[df['Outcome'] == 1]
diabetes_negative = df[df['Outcome'] == 0]

# Calculate the mean ages for the two groups
mean_age_diabetes_positive = diabetes_positive['Age'].mean()
mean_age_diabetes_negative = diabetes_negative['Age'].mean()

# Perform a hypothesis test (two-sample t-test) to compare the mean ages
_, p_value = stats.ttest_ind(diabetes_positive['Age'], diabetes_negative['Age'], equal_var=False)

# Determine the significance level
alpha = 0.05

# Print the results
print("Mean Age Comparison:")
print(f"Mean Age (Diabetes Positive): {mean_age_diabetes_positive:.2f}")
print(f"Mean Age (Diabetes Negative): {mean_age_diabetes_negative:.2f}")
print(f"P-value: {p_value}")
if p_value < alpha:
    print("There is a significant difference in mean ages between individuals with and without diabetes.")
else:
    print("There is no significant difference in mean ages between individuals with and without diabetes.")
